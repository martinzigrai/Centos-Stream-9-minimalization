1. krok - instalacia x servru a prehliadaca
dnf -y install xorg-x11-server-Xorg xinit firefox

2. krok - vytvorenie adresara ~/client 
mkdir ~/client

3. krok - vytvorenie bash skriptu v adresari ~/client 
vi ~/client/xClient.sh

toto treba dat do toho skriptu
	while [ true ]
	do
		firefox
	done

4. krok 
chmod +x ~/client/xClient.sh

5. krok - spustenie x servru
startx ~/client/xClient.sh -- :1 &
